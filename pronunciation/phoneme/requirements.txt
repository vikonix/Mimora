# Dependencies for the pronunciation/phoneme/ module (text-only phoneme engine).
# Install on top of the project root requirements:
#   pip install -r pronunciation/phoneme/requirements.txt
#
# Most deps overlap with pronunciation/acoustic/ (transformers/torch/librosa for the wav2vec2
# recognizer, phonemizer-fork for the espeak reference). The one addition over the
# acoustic core is panphon, used for the articulatory feature distance.
#
# NOTE: phonemizer needs espeak-ng. In the full app Kokoro/misaki registers the
# bundled espeakng_loader on import; standalone, the module registers it defensively
# (see speech._ensure_espeak) or falls back to a system espeak-ng on PATH.

# >=5.3 closes CVE-2026-4372 (config-injection RCE); kept in sync with the root
# requirements.txt. The markers below relax torch/transformers on Intel macOS
# (x86_64) - see the "Intel macOS note" in the root requirements.txt.
transformers>=5.3; platform_system != 'Darwin' or platform_machine != 'x86_64'
transformers>=4.44,<5; platform_system == 'Darwin' and platform_machine == 'x86_64'
torch>=2.0.0; platform_system != 'Darwin' or platform_machine != 'x86_64'
torch==2.2.2; platform_system == 'Darwin' and platform_machine == 'x86_64'
librosa>=0.10.0
# Use phonemizer-fork, NOT plain "phonemizer" (shares the import namespace; the fork
# adds EspeakWrapper.set_data_path() that Kokoro/misaki rely on).
phonemizer-fork>=3.3.0
# Articulatory feature distance for the phoneme substitution cost (the one dep the
# acoustic core does not need).
panphon>=0.20.0
# Intel macOS needs <2 for torch 2.2.2 (see root requirements.txt note).
numpy>=1.24.0; platform_system != 'Darwin' or platform_machine != 'x86_64'
numpy>=1.24.0,<2; platform_system == 'Darwin' and platform_machine == 'x86_64'
